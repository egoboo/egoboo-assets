// ZZ> This function makes an item fall to the floor when spawned

IfUsed
  tmpargument = 35    // Set attack speed
  SetReloadTime
  IfAmmoOut
    tmpargument = 3    // No ammo, play string sound
    tmpdistance = 11025
    PlaySound

IfSpawned
  MakeAmmoKnown
  tmpargument = ACTIONJB
  DoAction
  KeepAction
  MakeCrushValid

IfCrushed
  tmpx = selfx
  tmpy = selfy
  tmpdistance = selfz
  tmpargument = 4
  SpawnExactParticle
  SpawnExactParticle
  SpawnExactParticle
  SpawnExactParticle
  GoPoof
  tmpargument = 1
  SendMessageNear

// Make it lie on floor
IfDropped
  KeepAction

// Play the drop sound
IfHitGround
  tmpargument = 2
  tmpdistance = rand % 2047 + 7000
  PlaySound

// Tell them what they've won...
IfGrabbed
  SetTargetToWhoeverIsHolding
  IfTargetIsAPlayer
    tmpargument = 0
    SendMessageNear

// All done
End
